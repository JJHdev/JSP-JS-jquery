//변수 key에 저장한 후 출력
let text2 = "<ul>--아래는 for in문을 이용하여 배열의 인덱스번호 출력";

	text2+="<li>"+key+"</li>";
}

document.getElementById("demo2").innerHTML = text2;

  text1 += "<li>" + value + "</li>";

for/in- 객체의 속성을 반복. 
				for(변수 in 객체){}
for/of- 반복 가능한 객체의 값을 반복.

while- 지정된 조건이 참인 동안 코드 블록을 반복. while(조건){}
do/while- 또한 지정된 조건이 참인 동안 코드 블록을 반복 do{}while(조건)
